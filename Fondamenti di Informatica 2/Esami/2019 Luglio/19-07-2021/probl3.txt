Algoritmo ksort(List s, int k)
    int max_num = infinito
    List s = lista vuota
    while dimensione di res <= k do:
        int num = 0
        for elemento in s do:
            if elemento > num and elemento < max_num:
                num = elemento
        inserisci num in res
        max_num = num
    return res

costo = O(kn)
